# Page Content Types Template

> TypeScript interfaces for structured page content.

## Template

```typescript
/**
 * Page Content Types
 *
 * TypeScript interfaces for structured content that can be
 * converted to markdown. Used by the Beacon pack.
 */

/**
 * Represents a complete page's content structure.
 */
export interface PageContent {
  /** Page title - used in frontmatter and as H1 */
  title: string;

  /** Short description - used in frontmatter and as lead paragraph */
  description?: string;

  /** Last updated date in ISO format */
  lastUpdated?: string;

  /** Author name or identifier */
  author?: string;

  /** Tags for categorization */
  tags?: string[];

  /** Main content sections */
  sections: ContentSection[];

  /** Optional metadata for additional frontmatter fields */
  metadata?: Record<string, string | number | boolean>;
}

/**
 * Represents a section of content within a page.
 */
export interface ContentSection {
  /** Section heading text */
  heading?: string;

  /** Heading level (2-6, defaults to 2) */
  level?: number;

  /** Content type determines rendering */
  type?: ContentType;

  /** The actual content text */
  content: string;

  /** Language hint for code blocks */
  language?: string;
}

/**
 * Supported content types for sections.
 */
export type ContentType =
  | 'paragraph'   // Default, plain text paragraphs
  | 'list'        // Bullet list (items separated by newlines)
  | 'code'        // Code block with optional language
  | 'table'       // Markdown table format
  | 'blockquote'; // Quoted text

/**
 * Metadata that can be included in YAML frontmatter.
 */
export interface PageMetadata {
  title: string;
  description?: string;
  last_updated?: string;
  author?: string;
  tags?: string[];
  generated_at?: string;
  [key: string]: string | string[] | undefined;
}

/**
 * Result of parsing markdown back to structured content.
 */
export interface ParsedMarkdown {
  frontmatter: PageMetadata;
  content: string;
  sections: ContentSection[];
}

/**
 * Options for markdown generation.
 */
export interface MarkdownOptions {
  /** Include YAML frontmatter (default: true) */
  includeFrontmatter?: boolean;

  /** Include table of contents (default: false) */
  includeTableOfContents?: boolean;

  /** Maximum heading level to include in TOC (default: 3) */
  tocMaxLevel?: number;

  /** Add "Generated by" footer (default: false) */
  includeGeneratedFooter?: boolean;
}

/**
 * Helper type for creating page content from raw data.
 */
export type PageContentInput = Omit<PageContent, 'sections'> & {
  sections?: Array<Partial<ContentSection> & { content: string }>;
};

/**
 * Create a PageContent object with defaults applied.
 */
export function createPageContent(input: PageContentInput): PageContent {
  return {
    title: input.title,
    description: input.description,
    lastUpdated: input.lastUpdated,
    author: input.author,
    tags: input.tags,
    metadata: input.metadata,
    sections: (input.sections || []).map(section => ({
      type: 'paragraph' as ContentType,
      level: 2,
      ...section,
    })),
  };
}
```

## Placeholders

This template has no dynamic placeholders - it's used as-is.

## Usage Example

```typescript
import type {
  PageContent,
  ContentSection,
  ContentType,
  MarkdownOptions,
} from '@/lib/markdown/page-content';
import { createPageContent } from '@/lib/markdown/page-content';

// Type-safe page creation
const page: PageContent = {
  title: 'API Reference',
  description: 'Complete API documentation',
  lastUpdated: '2026-01-20',
  sections: [
    {
      heading: 'Authentication',
      level: 2,
      type: 'paragraph',
      content: 'All API requests require authentication.',
    },
    {
      heading: 'Example Request',
      level: 3,
      type: 'code',
      language: 'bash',
      content: 'curl -H "Authorization: Bearer $TOKEN" https://api.example.com',
    },
  ],
};

// Or use helper
const quickPage = createPageContent({
  title: 'Quick Start',
  description: 'Get started in 5 minutes',
  sections: [
    { content: 'Welcome to our platform.' },
    { heading: 'Installation', content: 'npm install our-package' },
  ],
});
```
